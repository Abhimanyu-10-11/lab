import {useState} from "react";
import './App.css';

function App() {
  const [studentList, setStudentList] = useState([
    {id: 1, name: "S1", percentage: 91},
    {id: 2, name: "S2", percentage: 81},
    {id: 3, name: "S3", percentage: 33},
    {id: 4, name: "S4", percentage: 46},
    {id: 5, name: "S5", percentage: 87},
    {id: 6, name: "S6", percentage: 93},
    {id: 7, name: "S7", percentage: 72},
    {id: 8, name: "S8", percentage: 60},
    {id: 9, name: "S9", percentage: 25},
    {id: 10, name: "S10", percentage: 50}
  ]);
  const [distinctionCount, setDistinctionCount] = useState(0);
  const [firstClassCount, setFirstClassCount] = useState(0);

  const updateStudentList = (id, newPercentage) => {
    setStudentList(studentList.map((student) =>
      student.id === id ? {...student, percentage: newPercentage} : student
    ));
  };

  const updateCount = () => {
    setDistinctionCount(studentList.filter((student) =>
      student.percentage >= 85
    ).length);
    setFirstClassCount(studentList.filter((student) =>
      student.percentage >= 60 && student.percentage < 85
    ).length);
  }

  return (
    <div className="App">
      <h3>Meritorious Student Counter</h3>
      <div className="studentList">
        {studentList.map((student) => (
          <div key={student.id}>
            <span>{student.name}: </span>
            <input type="number" value={student.percentage} onChange={(e) =>
              updateStudentList(student.id, Number(e.target.value))
            }></input>{" %"}
          </div>
        ))}
      </div>
      <button onClick={updateCount}>Calculate</button>
      <div className="Results">
        <p>Distinction holding student count: {distinctionCount}</p>
        <p>First class holding student count: {firstClassCount}</p>
      </div>
    </div>
  );
}

export default App;

import React, {useState, useEffect} from 'react';
import './App.css';

function App() {
  const [inputEmail, setInputEmail] = useState("");
  const [validEmail, setValidEmail] = useState(false);
  const [feedback, setFeedback] = useState("");

  const validateEmail = (email) => {
    const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if(regex.test(email)){
      setValidEmail(true);
      setFeedback("This is a valid email");
    }
    else{
      setValidEmail(false);
      setFeedback("This email is invalid");
    }
  }

  useEffect(() => {
    validateEmail(inputEmail);
  }, [inputEmail]);

  return (
    <div className="App">
      <h2>Email Validator</h2>
      <label htmlFor="inputEmail">Email: </label>
      <input type="text" id="inputEmail" value={inputEmail}
      onChange={(e) => {
        setInputEmail(e.target.value);
      }}/>

      <div style={{color: validEmail ? 'green': 'red'}}>{feedback}</div>
    </div>
  );
}

export default App;

NotificationService.java
package com.example.demo;

import org.springframework.stereotype.Component;

@Component
public class NotificationService {
    private final MessageService messageService;

    public NotificationService(MessageService messageService) {
        this.messageService = messageService;
    }

    public void notifyUser(String message, String recipient) {
        messageService.sendMessage(message, recipient);
    }
}

DependencyInjectionApplication
package com.example.demo;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.ApplicationContext;

@SpringBootApplication
public class DependencyInjectionApplication {
    public static void main(String[] args) {
        ApplicationContext context = SpringApplication.run(DependencyInjectionApplication.class, args);
        NotificationService notificationService = context.getBean(NotificationService.class);
        notificationService.notifyUser("Hello, World!", "user@example.com");
    }
}
